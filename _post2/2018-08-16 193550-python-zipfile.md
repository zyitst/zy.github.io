##使用zipfile模块
    import os, zipfile
	rootdir=os.getcwd()
	l=os.listdir(rootdir)		#获取当前路径下所有文件、文件夹

	with zipfile.ZipFile('test.zip', mode='w') as zipf:
	    for i in range(len(l)):
	        path=os.path.join(rootdir,l[i])
	        if os.path.isfile(path):	#如果是文件而不是目录则写入压缩文件
	            zipf.write(l[i])
